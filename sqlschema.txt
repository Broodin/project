CREATE TABLE APPLICATION(
	ID VARCHAR(50) PRIMARY KEY,
	NAME VARCHAR(200) NOT NULL,
	USER VARCHAR(20) NOT NULL,
	QUEUE VARCHAR(30),
	START_TIME VARCHAR(30),
	FINISH_TIME VARCHAR(30),
	STATE VARCHAR(10),
	URL VARCHAR(200),
	ELAPSED_TIME VARCHAR(20)
);





CREATE TABLE HEURISTIC(
	JOB_ID VARCHAR(50),
	H_NAME VARCHAR(50),
	SCORE INT,
	SEVERITY VARCHAR(5),
	FOREIGN KEY(JOB_ID) REFERENCES APPLICATION(ID)

);
